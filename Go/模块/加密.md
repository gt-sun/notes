## `sha1/MD5`加密

```go
import (
    "crypto/sha1"
    "fmt"
    "io"
    "log"
)

func main() {
    // 法1
    hasher := sha1.New() // md5.New()
    io.WriteString(hasher, "test")
    b := []byte{}
    fmt.Printf("法1：Result is %x\n", hasher.Sum(b))
    fmt.Printf("法1：Result is %d\n", hasher.Sum(b))

    //法2
    hasher.Reset()
    data := []byte("test")
    n, err := hasher.Write(data)
    if n != len(data) || err != nil {
        log.Printf("Hash write error: %v / %v", n, err)
    }
    fmt.Printf("法2：Result is %x\n", hasher.Sum(b))
}
```
